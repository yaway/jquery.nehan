jquery.nehan demo

1x1 vertical rl (progress bar version)

$(".1x1-vert-progress").nehan({
  direction:"vert",
  width:400,
  height:300,
  fontSize:16,
  rowCount:1,
  colCount:1,
  spacingSize:16,
  readerElements:["screen", "pager"],
  pagerElements:["progress", "left-next", "right-prev"]
});
